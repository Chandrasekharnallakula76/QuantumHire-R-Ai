1. Is there an API for scanning the face and displaying vitals?
NO, there is no third-party cloud API or paid external service used to scan the face or calculate the vitals.

Everything is done 100% locally in the user's browser for privacy and speed.

How to add this to your new project:
To implement face tracking and vitals calculation in your new project, you do not need to register for any API keys. You only need to copy the local setup:

typescript
const ELEVENLABS_API_KEY = "sk_05dc65f617d450f4c90bdbc29dba3925e05533504478d06c";
const ELEVENLABS_VOICE_ID = "7FroLDTDG92jPfUW6BlQ";
const ELEVENLABS_MODEL_ID = "eleven_turbo_v2_5";
let currentAudio: HTMLAudioElement | null = null;
const elevenLabsSpeak = async (
  text: string,
  onStart: () => void,
  onEnd: () => void,
): Promise<void> => {
  try {
    if (currentAudio) {
      currentAudio.pause();
      currentAudio = null;
    }
    onStart(); // Start avatar animation/talking state
    // Request speech audio stream from ElevenLabs
    const response = await fetch(
      `https://api.elevenlabs.io/v1/text-to-speech/${ELEVENLABS_VOICE_ID}/stream`,
      {
        method: "POST",
        headers: {
          Accept: "audio/mpeg",
          "Content-Type": "application/json",
          "xi-api-key": ELEVENLABS_API_KEY,
        },
        body: JSON.stringify({
          text: text,
          model_id: ELEVENLABS_MODEL_ID,
          voice_settings: {
            stability: 0.5,
            similarity_boost: 0.75,
            style: 0.0,
            use_speaker_boost: true,
          },
          optimize_streaming_latency: 4, // 4 yields the absolute lowest latency
        }),
      },
    );
    if (!response.ok) {
      throw new Error(`ElevenLabs API error: ${response.status}`);
    }
    const audioBlob = await response.blob();
    const audioUrl = URL.createObjectURL(audioBlob);
    
    const audio = new Audio(audioUrl);
    currentAudio = audio;
    audio.preload = "auto";
    audio.onended = () => {
      URL.revokeObjectURL(audioUrl);
      currentAudio = null;
      onEnd(); // Stop avatar animation/talking state
    };
    audio.onerror = () => {
      URL.revokeObjectURL(audioUrl);
      currentAudio = null;
      onEnd();
    };
    await audio.play();
  } catch (error) {
    console.error("ElevenLabs TTS error:", error);
    
    // Fallback: If ElevenLabs fails/runs out of credits, use browser SpeechSynthesis
    if ("speechSynthesis" in window) {
      window.speechSynthesis.cancel();
      const utterance = new SpeechSynthesisUtterance(text);
      utterance.onstart = onStart;
      utterance.onend = onEnd;
      utterance.onerror = () => onEnd();
      window.speechSynthesis.speak(utterance);
    } else {
      onStart();
      setTimeout(onEnd, Math.max(1500, text.length * 40));
    }
  }
};
